CREATE TABLE requests
(
  doc_id TINYINT UNSIGNED NOT NULL,
  student_id INTEGER UNSIGNED NOT NULL,
  name_ar VARCHAR(80) CHARACTER SET utf8 NOT NULL,
  name_en VARCHAR(80) CHARACTER SET utf8,
  address VARCHAR(100) CHARACTER SET utf8,
  reason VARCHAR(200) CHARACTER SET utf8,
  mobile_number CHAR(11) CHARACTER SET utf8,
  year BIT(3),
  department CHAR(10) CHARACTER SET utf8,
  birth_date DATE,
  PRIMARY KEY(doc_id, student_id),
  INDEX(student_id)
);
